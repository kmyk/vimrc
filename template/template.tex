% -   いい感じに参照できる (cleverref)
% -   冒頭に改行 末尾にdiamond (amsthm)
\newcommand*{\newqedtheorem}[2]{
    \newenvironment{#1}[1][]{
        \begin{#2}[##1]
            \leavevmode
            \newline
            \renewcommand{\qedsymbol}{$$\diamond$$}
            \pushQED{\qed}
    }{
            \qedhere
            \popQED
        \end{#2}
    }
}

\newqedtheorem{theorem}{theorem*}

    \newtheorem{#1*}[theorem*]{#2}
    \Crefname{#1*}{#2}{#2}
    \crefname{#1*}{#2}{#2}
    \newqedtheorem{#1}{#1*}
}
